create table HRISHI.SALE_ITEM_DETAIL
(
 item_sln NUMBER(9)  CONSTRAINT item_sale_sln_fk REFERENCES HRISHI.SALE_DETAIL(SALE_SLN) 
                     ON DELETE CASCADE,
 item_name VARCHAR2(20) CONSTRAINT item_name_nn NOT NULL,
 qty      NUMBER(10),
 ppu      NUMBER(12,2),
 total    NUMBER(12,2))
PCTFREE 5
PCTUSED 30
INITRANS 10
MAXTRANS 15
TABLESPACE SAN
STORAGE(INITIAL 50K
        NEXT 50K
        MAXEXTENTS 10
        PCTINCREASE 20)        

/
